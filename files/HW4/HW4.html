Tuketim\(Tarih = as.POSIXct(strptime(Tuketim\)Tarih, format = %d.%m.%Y)) Tuketim\(`Tuketim Miktari (MWh)` <- as.numeric(Tuketim\)Tuketim Miktari (MWh))
